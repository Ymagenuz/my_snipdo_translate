$pythonExe = "C:\Users\EDY\AppData\Local\Programs\Python\Python311\pythonw.exe"
$scriptPath = "D:\test\my_snipdo_translate\gemini_translate.pyw"

# ===== 在这里填写你的 API Key =====
$apiKey = ""

$inputText = $args -join " "

if ([string]::IsNullOrWhiteSpace($inputText)) {
    exit
}

if (!(Test-Path $pythonExe)) {
    exit
}

if (!(Test-Path $scriptPath)) {
    exit
}

# 如果这里填了 Key，就注入当前 PowerShell 进程环境变量，子进程 pythonw 会继承。
# 如果留空，则交给 Python 界面在翻译前手动输入。
if (![string]::IsNullOrWhiteSpace($apiKey)) {
    $env:GPTSAPI_API_KEY = $apiKey
}

$tempFile = Join-Path $env:TEMP ("snipdo_gptsapi_" + [guid]::NewGuid().ToString() + ".txt")
[System.IO.File]::WriteAllText($tempFile, $inputText, [System.Text.UTF8Encoding]::new($false))

Start-Process -FilePath $pythonExe -ArgumentList @(
    "`"$scriptPath`"",
    "--file",
    "`"$tempFile`""
) -WindowStyle Hidden
